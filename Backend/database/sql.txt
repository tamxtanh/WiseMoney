create table
  public."Image" (
    id integer generated by default as identity,
    name text null default ''::text,
    description text null default ''::text,
    url text null default ''::text,
    constraint Image_duplicate_pkey primary key (id)
  ) tablespace pg_default;
  
create table
  public."User" (
    id integer generated by default as identity,
    name text null,
    username text not null,
    email text null,
    password text null,
    phone text null,
    initial_balance double precision null default '0'::double precision,
    created_date date null,
    status text null,
    avatar integer null,
    constraint User_pkey primary key (id),
    constraint User_username_key unique (username),
    constraint public_User_avatar_fkey foreign key (avatar) references "Image" (id) on delete set null
  ) tablespace pg_default;
